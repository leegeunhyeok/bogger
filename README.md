```javascript
console.log(...log.debug('We must catch the bugs'))

console.log(...log.success('Nice! happy :)'))
console.log(...log.warning('Watch out!'))
console.log(...log.error('That\'s so dangerous'))
console.log(...log.critical('** DEAD **'))
```
